<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Cookies CRUD operations :</title>
</head>

<body>
  <center>
    <h1>Cookies CRUD operations :</h1>
    <input type="text" id="newCookiesKey" placeholder="Enter Key of Cookie">
    <input type="text" id="newCookiesValue" placeholder="Enter Value of Cookie">
    <button onclick="createCookies()">Create Cookies</button> <br> <br>

    <input type="text" id="updateCookiesKey" placeholder="Enter Key of Cookie">
    <input type="text" id="updateCookiesValue" placeholder="Enter Value of Cookie">
    <button onclick="updateCookies()">Update Cookies</button> <br> <br> <br>
    <input type="text" id="deleteCookiesKey" placeholder="Enter Key of Cookie">
    <button onclick="deleteCookies()">Delete Cookies</button> <br> <br>
    <button onclick="readCookies()">Read Cookies</button>
    <p id="display"></p>
  </center>
  <script>
    let display = document.getElementById("display");
    // Variables for user input
    let newCookiesKey = document.getElementById("newCookiesKey");
    let newCookiesValue = document.getElementById("newCookiesValue");
    // variables for update cookies
    let updateCookiesKey = document.getElementById("updateCookiesKey");
    let updateCookiesValue = document.getElementById("updateCookiesValue");
    // variables for delete cookies
    let deleteCookiesKey = document.getElementById("deleteCookiesKey");
    // create cookies (Function Callback)
    function createCookies() {
      document.cookie = newCookiesKey.value + "=" + newCookiesValue.value;
      display.innerHTML = "Cookies Created";
    }
    function readCookies() {
      let cookies = document.cookie;
      if (cookies) {
        display.innerHTML = "Cookies found : " + cookies;
      } else {
        display.innerHTML = "Cookies not found";
      }
    }
    // update cookies
    function updateCookies() {
      let temp = updateCookiesKey.value + "=" + updateCookiesValue.value;
      document.cookie = temp;
      display.innerHTML = "Cookies Updated";
    }
    // delete cookies
    function deleteCookies() {
      let temp = `${deleteCookiesKey.value}=; expires=Thu, 01 Jan 1970 00:00:00 GMT;`;
      document.cookie = temp;
      display.innerHTML = "Cookies Deleted";
    }
  </script>
